131 HBRUSH CHyperlinkLabel::CtlColor(CDC* pDC, UINT /*nCtlColor*/)
132 {
133  // Declare variables
134  CString sWindowText;
135  CSize size;
136 
137  // Get window text
138  GetWindowText(sWindowText);
139 
140  // Resize control according to text length
141  if (m_bAutoResize)
142  {
143  size = pDC->GetOutputTextExtent(sWindowText);
144  SetWindowPos(NULL, 0, 0, size.cx, size.cy, SWP_NOZORDER | SWP_NOMOVE | SWP_NOREDRAW);
145  }
146 
147  // Be sure we are getting notified
148  ModifyStyle(0, SS_NOTIFY, 0);
149 
150  // Check which color to draw
151  if (m_sHyperlink.IsEmpty())
152  {
153  // Use normal color
154  m_clrCurrent = m_clrNormal;
155  }
156  else
157  {
158  // Hover or link color?
159  if (m_bHover)
160  {
161  m_clrCurrent = m_clrHover;
162  }
163  else
164  {
165  // Set new color to normal link
166  m_clrCurrent = m_clrLink;
167  }
168  }
169 
170  // Set text color
171  pDC->SetTextColor(m_clrCurrent);
172 
173  // Transparent text.
174  pDC->SetBkMode(TRANSPARENT);
175 
176  // Return value
177  return (HBRUSH)GetStockObject(NULL_BRUSH);
178 }
